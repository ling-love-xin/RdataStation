-- 查询编辑器 - 即时查询模式 SELECT u.id, u.name, u.email, COUNT(o.id) AS order_count, SUM(o.total) AS total_spent FROM users u LEFT JOIN orders o ON u.id = o.user_id WHERE u.status = 'active' AND u.created_at >= '2025-01-01' GROUP BY u.id, u.name, u.email HAVING COUNT(o.id) > 0 ORDER BY total_spent DESC LIMIT 100;

-- 分析编辑器 - DuckDB 联邦查询模式 -- 已 ATTACH: prod_users (MySQL), analytics (PG) CREATE OR REPLACE TABLE user_360 AS WITH monthly_orders AS ( SELECT user_id, DATE_TRUNC('month', created_at) AS month, SUM(amount) AS revenue, COUNT(*) AS orders FROM prod_users.orders WHERE created_at >= '2025-01-01' GROUP BY 1, 2 ), user_behavior AS ( SELECT user_id, AVG(session_duration) AS avg_session, COUNT(DISTINCT page_id) AS unique_pages FROM analytics.sessions GROUP BY user_id ) SELECT u.name, mo.month, mo.revenue, ub.avg_session, LAG(mo.revenue) OVER w AS prev_revenue FROM prod_users.users u JOIN monthly_orders mo ON u.id = mo.user_id JOIN user_behavior ub ON u.id = ub.user_id WINDOW w AS (PARTITION BY u.id ORDER BY mo.month) ORDER BY mo.month, revenue DESC;

# 代码编辑器 - 通用文件编辑模式 # 支持: .rs .ts .tsx .js .py .go .java .c .cpp import pandas as pd import duckdb def export_user_report(db_path: str, output: str) -> pd.DataFrame: """导出用户分析报告为 CSV""" conn = duckdb.connect(db_path) df = conn.execute(""" SELECT name, month, revenue, avg_session FROM user_360 WHERE revenue > 5000 ORDER BY revenue DESC """).df() df.to_csv(output, index=False) conn.close() return df if __name__ == '__main__': result = export_user_report( 'analytics/data.duckdb', 'exports/report_2025.csv' ) print(f"导出完成: {len(result)} 行")
